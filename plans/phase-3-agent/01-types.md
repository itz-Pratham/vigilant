# Phase 3 — Agent Types

**File:** `src/agent/types.ts`

## All Agent Types

```typescript
// ── Stage Machine ─────────────────────────────────────────────────────

/**
 * Every possible stage in a session's lifecycle.
 * Stages move forward only. The exception is 'blocked' which can recover
 * back to the previous stage after human intervention or auto-retry.
 *
 * Terminal stages: merged, skipped, closed
 * Blocking stage: blocked (stores previous stage in dataCollected.previousStage)
 */
export type IssueStage =
  | 'discovered'           // watcher created the session — no agent work yet
  | 'investigating'        // agent is running the agentic loop
  | 'planning'             // agent is generating the structured plan
  | 'awaiting_self_review' // plan generated — self-review loop about to start
  | 'self_reviewing'       // self-review loop is actively running (max 3 iterations)
  | 'awaiting_approval'    // self-review passed — waiting for Gate 1
  | 'executing'            // Gate 1 approved — executor is running
  | 'pr_created'           // PR is open — waiting for CI
  | 'awaiting_merge'       // CI passed — waiting for Gate 2
  | 'merged'               // merged — terminal
  | 'skipped'              // human skipped at Gate 1 — terminal
  | 'closed'               // human closed PR at Gate 2 — terminal
  | 'blocked';             // something went wrong — needs attention

export type Severity = 'LOW' | 'MEDIUM' | 'HIGH' | 'CRITICAL';

// ── Plan ─────────────────────────────────────────────────────────────

/**
 * A single file change to be applied by the executor.
 */
export type FileChange = {
  /** Repo-relative file path, e.g. "src/checkout/payment.ts" */
  path: string;
  /** Human-readable description of what this change does */
  description: string;
  /**
   * The existing code that is being replaced (for display in Gate 1).
   * Should be the exact code snippet, not a diff.
   * If the change is purely additive, this is an empty string.
   */
  before: string;
  /**
   * The replacement code (for display in Gate 1).
   * Should be the exact new code snippet.
   */
  after: string;
  /** Approximate line number where the change occurs, for display context */
  lineHint?: number;
  /**
   * If true, this is a new file to be created, not an edit to an existing file.
   * The executor uses this to decide between PUT (update) and POST (create).
   */
  isNewFile?: boolean;
};

/**
 * The complete investigation result and fix plan.
 * Generated by plan-generator.ts after the agentic loop completes.
 * Displayed to the human at Gate 1.
 */
export type Plan = {
  /** One-sentence summary of the issue */
  summary: string;
  severity: Severity;
  /** 0.0–1.0 confidence that this is a real issue requiring a fix */
  confidence: number;
  /** Multi-sentence explanation of why this is a problem and where it comes from */
  rootCause: string;
  /** Ordered list of code changes to apply. Minimum 1. */
  changes: FileChange[];
  /** Branch name to create. Format: vigilant/fix/{issueType-kebab}-{shortRef} */
  branchName: string;
  /** PR title. Format: fix({issueType}): {description} [vigilant] */
  prTitle: string;
  /** Full PR body in markdown. Includes root cause, changes, test suggestions, session ID. */
  prBodyMarkdown: string;
  /** 1–3 suggested test cases to add for this fix */
  testSuggestions: string[];
};

// ── Session ───────────────────────────────────────────────────────────

/**
 * The complete state of a vigilant investigation session.
 * Persisted to SQLite after every agentic loop iteration.
 * This is the single source of truth for a session's lifecycle.
 */
export type IssueSession = {
  /** Unique session identifier. Immutable once created. */
  sessionId: string;
  repoOwner: string;
  repoName: string;
  /** Active domain pack at time of detection */
  domain: string;
  /** Issue type, e.g. "MISSING_IDEMPOTENCY". From the domain pack's PatternRule. */
  issueType: string;
  stage: IssueStage;
  severity: Severity;
  /** 0.0–1.0. Starts from PatternRule.confidenceScore, may be updated by agent. */
  confidence: number;
  /** Where the issue was found. Format: "PR#47" | "file:path:line" | etc. */
  sourceRef: string;
  /** Lines of evidence from the scanner that triggered this session */
  evidence: string[];

  // ── Agent loop progress ─────────────────────────────────────────────
  /** How many NeuroLink generate() calls have been made in this session */
  iterationCount: number;
  /**
   * Agent's self-reported progress toward completing the investigation.
   * 0.0 = just started, 1.0 = complete.
   * Extracted from the model's response after each iteration.
   */
  goalProgress: number;
  /** Key findings accumulated across all iterations */
  keyFindings: string[];
  /**
   * Structured data collected by agent tools (file contents, search results, etc.)
   * Not shown to the user directly — used by plan-generator.
   */
  dataCollected: Record<string, unknown>;
  /** How many consecutive iterations without goalProgress improvement */
  stallCount: number;

  // ── Plan ─────────────────────────────────────────────────────────────
  /** Populated once stage reaches 'awaiting_approval' */
  plan: Plan | null;

  // ── Executor progress ─────────────────────────────────────────────
  branchName: string | null;
  prNumber: number | null;
  prUrl: string | null;
  prHeadSha: string | null;
  ciStatus: 'pending' | 'running' | 'passed' | 'failed' | null;
  /** Last completed executor step — for crash recovery */
  executorStep: 'branch_created' | 'files_written' | 'pr_created' | null;
  /** How many self-review iterations have been run (max 3) */
  selfReviewCount: number;

  // ── Blocking ──────────────────────────────────────────────────────
  /** Human-readable reason why this session is blocked */
  blockerReason: string | null;

  // ── Metadata ─────────────────────────────────────────────────────
  /**
   * Increments when a new session is created for the same issueType+sourceRef
   * after a previous session was resolved. Allows full history.
   */
  runNumber: number;
  createdAt: string;
  updatedAt: string;
};

// ── Agent Tools ───────────────────────────────────────────────────────

/**
 * A tool available to the agent during the investigation loop.
 * Tools are passed to NeuroLink and called by the model autonomously.
 */
export type AgentToolName =
  | 'readFile'
  | 'searchCode'
  | 'ragSearch'
  | 'readPRDiff'
  | 'searchWeb'
  | 'readGitHistory'      // reads last N commits + merged PRs for a file/repo
  | 'readTeamDecisions'   // reads decision.md, adr/, docs/ from the repo
  | 'getCurrentTime'
  | 'sequentialThinking';

export type AgentToolCall = {
  name: AgentToolName;
  parameters: Record<string, unknown>;
};

export type AgentToolResult = {
  toolName: AgentToolName;
  success: boolean;
  result: unknown;
  error?: string;
};
```
